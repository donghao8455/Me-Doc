// GetEmail decrypts and returns the email address
 func (p *Profile) GetEmail() (string, error) {
     decoded, err := base64.StdEncoding.DecodeString(p.EncryptedEmail)
     if err != nil {
         return "", err
    }
     return string(decoded), nil
 }
